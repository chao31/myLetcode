### 实现并发
```js
function task([promise1, promise2...]) {
}
```

实现一个写个任务调度器，同时并发最多执行 3 个任务，用 js 实现

```js
async function task(promises) {
  const results = [];

  // 循环执行任务
  while (promises.length) {
    // 获取前三个任务
    const tasks = promises.splice(0, 3);

    // 并发执行任务
    const resultsPromise = Promise.allSettled(tasks.map((task) => task()));

    // 等待所有任务执行完毕
    const currentResults = await resultsPromise;

    // 将当前结果加入总结果中
    results.push(...currentResults);
  }

  return results;
}
```

### 实现倒计时

```js
import React, { useState, useEffect } from "react";

function Countdown({ targetDate, onCountdownEnd }) {
  const [countdown, setCountdown] = useState(targetDate - Date.now());

  useEffect(() => {
    const intervalId = setInterval(() => {
      const remainingTime = targetDate - Date.now();
      setCountdown(remainingTime);
      if (remainingTime <= 0) {
        clearInterval(intervalId);
        onCountdownEnd();
      }
    }, 1000);
    return () => clearInterval(intervalId);
  }, [targetDate, onCountdownEnd]);

  const seconds = Math.floor(countdown / 1000) % 60;
  const minutes = Math.floor(countdown / 1000 / 60) % 60;
  const hours = Math.floor(countdown / 1000 / 60 / 60) % 24;
  const days = Math.floor(countdown / 1000 / 60 / 60 / 24);

  return (
    <div>
      {days}d {hours}h {minutes}m {seconds}s
    </div>
  );
}

```
