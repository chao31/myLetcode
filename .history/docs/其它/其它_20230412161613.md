### 使用 js 实现斐波那契数列

前 2 个加起来是第 3 个

```js
1,1,2,3,5,8,13,21,34....
```

```js
function fibonacci(n) {
    if(n < 2) return n;

    return fibonacci(n -1) + fibonacci(n - 2)
}
```
用 js 暴力

```js
function fibonacciIterative(n) {
  if (n <= 1) {
    return n;
  }
  let prev = 0;
  let curr = 1;
  for (let i = 2; i <= n; i++) {
    const temp = prev + curr;
    prev = curr;
    curr = temp;
  }
  return curr;
}
```
