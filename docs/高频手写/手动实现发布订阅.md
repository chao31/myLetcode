手写一个基于发布订阅模式的 js 事件处理中心 (EventEmitter)

## 简版的实现

```js
class EventEmitter {
  constructor() {
    this.events = {};
  }

  on(type, fn) {
    this.events[type] || (this.events[type] = []);
    this.events[type].push(fn);
  }

  emit(type, ...args) {
    const arr = this.events[type] || [];
    arr.forEach(fn => fn(...args));
  }

  off(type) {
    // delete this.events[type];
    this.events[type] = this.events[type].filter(element => element !== fn);
  }
}

// 测试
// let aa = new EventEmitter();

// aa.on('click', () => {console.log(1)});
// aa.emit('click'); // 1 

// aa.off('click');
// aa.emit('click'); // undefined
```

## 添加 once 方法

```js
class EventEmitter {
  constructor() {
    this.events = {};
  }

  on(type, fn) {
    this.events[type] || (this.events[type] = []);
    this.events[type].push(fn);
  }

  once = (type, fn) => {
    const onceFn = (...args) => {
      fn(...args);
      this.off(type, onceFn);
    }
    this.on(type, onceFn);
  }

  emit(type, ...args) {
    const arr = this.events[type] || [];
    arr.forEach(fn => fn(...args));
  }

  off(type, fn) {
    this.events[type] = this.events[type].filter(element => element !== fn);
  }
}

// 测试
// let aa = new EventEmitter();

// let myFn = (num) => {console.log(num)};
// aa.on('click', myFn);
// aa.emit('click', 1); // 1 
// aa.emit('click', 2); // 2

// aa.off('click', myFn);
// aa.emit('click', 1); // 

// aa.once('click', myFn);
// aa.emit('click', '执行 1 次'); // 1 
// aa.emit('click', '再执行 1 次'); // 
```
